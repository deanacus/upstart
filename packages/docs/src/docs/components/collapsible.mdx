import { useState } from 'react';
import { PropsTable, DemoBox } from '../../components/Documentation';
import { Collapsible, Button, Stack } from '@deanacus/upstart';

export const Demo = () => {
  const [collapsed, toggleCollapsed] = useState(true);
  return (
    <div style={{
      margin: '3.2rem 0',
      border: '1px solid #ccc',
      borderRadius: '.5rem',
      padding: '1.6rem',
    }}>
    <Stack>
      <div>
        <Button 
          handleClick={() => toggleCollapsed(!collapsed)}
          size="sm"
          isOutline
        >
          Toggle
        </Button>
      </div>
      <Collapsible collapsed={collapsed}>
        <DemoBox>Item</DemoBox>
      </Collapsible>
      </Stack>
    </div>
  )
}

Collapsible is a simple wrapper for children that allows itself to be either
collapsed or expanded. It does not provide any method of toggling it's own
state, instead it relies on the props passed to it to determine this.

```jsx
<Collapsible collapsed>
  <DemoBox>Item</DemoBox>
</Collapsible>
<Collapsible>
  <DemoBox>Item</DemoBox>
</Collapsible>
```
